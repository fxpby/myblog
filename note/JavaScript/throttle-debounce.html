const throttle = (func, wait = 1000) => {
  let lastTime = 0
  let timer = null

  return (...args) => {
    if (timer) {
      clearTimeout(timer)
      timer = null
    }

    let nowTime = +new Date()

    const remainWaitTime = wait - (nowTime - lastTime)

    if (remainWaitTime <= 0) {
      lastTime = nowTime
      func(...args)
    } else {
      timer = setTimeout(() => {
        lastTime = +new Date()
        func(...args)
        timer = null
      }, remainWaitTime)
    }
  }
}
