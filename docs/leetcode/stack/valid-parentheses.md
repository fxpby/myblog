var isValid = function(s) {
  if (s.length % 2) {
    return false
  }
  
  const stack = []
  const pairObj = {
    ')': '(',
    '}': '{',
    ']': '['
  }

  for(let ch of s) {
    if ([')', '}', ']'].includes(ch)) {
      if (!stack.length || stack[stack.length - 1] !== pairObj[ch]) {
        return false
      }
      stack.pop()
    } else {
      stack.push(ch)
    }
  }

  return !stack.length
};

var isValid = function(s) {
  let len = s.length
  while (len && len % 2 === 0) {
    s = s.replace('()', '').replace('{}', '').replace('[]', '')
    if (s.length === len) {
      return false
    }
    len = s.length
  } 
  return len === 0
};
